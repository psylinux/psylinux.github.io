1. Static compiling

1

gcc -static cve_2016_0728.c -o cve_2016_0728 -lkeyutils -Wall

2. Cross compiling a C code from Linux to Windows

2.1. Install mingw-w64 cross-compiler:

1

apt-get install mingw-w64

2.2. To compile a Windows PE Executable 32bit:

1
2

i686-w64-mingw32-gcc 271.c -o 271
x86_64-w64-mingw32-gcc-win32 271.c -o 271

2.3. Run the binary file with wine:

1

wine file.exe

3. Compiling with debug symbols

Debug Symbol File Types:

  1. DWARF 2
  2. COFF
  3. XCOFF
  4. STABS

3.1. GCC use the -g option to define the type:

1

gcc -ggdb source.c -o prog_with_symbols

3.2. Stripping symbols off the binary

1

objcopy --strip-debug --strip-unneeded prog_not_stripped prog_stripped

3.3. Ripping Symbols off the binary

1

objcopy --only-keep-debug rip_from_binary debug_file

#include <stdio.h>

int main(void){
    int var = 5;
    int *p;

    printf("Address of var is: %p\n", &var);
    printf("Address of p is: %p\n", &p);

    p = &var;
    printf("Content of p is: %p\n", p);
    printf("Content wich p is pointing is: %i\n", *p);
    printf("Content of var is: %i\n", var);
    printf("Address of var is: %p\n", &var);
    printf("Address of p is: %p\n", &p);

    printf("*********************\n");
    *p = 100;
    printf("%i\n", var);
    printf("%i\n", *p);
    printf("Content of p is: %p\n", p);
    printf("Address of var is: %p\n", &var);
    printf("Address of p is: %p\n", &p);

    printf("*********************\n");
    p = 300;
    printf("%i\n", var);
    printf("%i\n", p);
    printf("Content of p is: %p\n", p);
    printf("Address of var is: %p\n", &var);
    printf("Address of p is: %p\n", &p);

    printf("*********************\n");
    printf("Final content of var: %i\n", var);
    printf("Content of p is: %p\n", p);
    return 0;
}
